function GetUsedUnitList(list: Tstringlist): string;
var
index: integer;
charcount: integer;
s: string;
begin
if (list <> nil) and (list.Count <> 0) then
begin
Result := 'Uses ' + list[0];
charcount := length(result);
for index := 1 to list.Count - 1 do
begin
s := list[index];
inc(charcount, length(s));
if charcount < 80 then
Result := Result + ', ' + s
else
begin
Result := Result + ', ' + NewLine + s;
charcount := 0;
end;
end;
Result := Result + ';';
end
else
Result := '';
end; {GetUsedUnitList}
